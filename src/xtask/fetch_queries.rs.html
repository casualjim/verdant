8pub fn run() -> Result<()> {
9    static QUERIES_DIR: Lazy<PathBuf> = Lazy::new(|| crate::WORKSPACE_DIR.join("queries"));
10    for lang in fs::read_dir(&*QUERIES_DIR)? {
11        let lang = lang?;
12        if lang.path().is_file() {
13            continue;
14        }
15        let lang_name = lang.file_name().to_string_lossy().into_owned();
16
17        for file in fs::read_dir(lang.path())? {
18            let file = file?;
19
20            let old_query = fs::read_to_string(file.path())?;
21            let filename = file.file_name().to_string_lossy().into_owned();
22
23            let Some((_, host, user, repo, branch, path)) = regex_captures!(
24                r"^;; Forked from https://(github|gitlab)\.com/([^/]*)/([^/]*)/(?:-/)?(?:blob|tree)/([^/]*)/([^?#\n]*)",
25                &old_query
26            ) else {
27                println!("\x1b[1;33mwarning:\x1b[22m {lang_name}/{filename} does not specify a fork source\x1b[0m");
28                continue;
29            };
30
31            let url = match host {
32                "github" => {
33                    format!("https://raw.githubusercontent.com/{user}/{repo}/{branch}/{path}")
34                }
35                "gitlab" => format!("https://gitlab.com/{user}/{repo}/-/raw/{branch}/{path}"),
36                _ => unreachable!("the regex only allows above options"),
37            };
38            println!("fetching new {lang_name}/{filename} from {url}");
39            let res = reqwest::blocking::get(&url).with_context(|| "query request failed")?;
40
41            let query = match res.status().is_success() {
42                true => res.text()?,
43                false => {
44                    // If github and 404, try flipping master <-> main branch
45                    if host == "github" && res.status() == 404 {
46                        let fallback_branch = if branch == "master" { "main" } else { "master" };
47                        let fallback_url = format!("https://raw.githubusercontent.com/{user}/{repo}/{fallback_branch}/{path}");
48                        println!("trying fallback: {fallback_url}");
49                        let fallback_res = reqwest::blocking::get(&fallback_url)
50                            .with_context(|| "fallback query request failed")?;
51                        if fallback_res.status().is_success() {
52                            fallback_res.text()?
53                        } else {
54                            bail!(
55                                "query request returned non-success status code: {} (also tried fallback: {})",
56                                res.status(),
57                                fallback_res.status()
58                            )
59                        }
60                    } else {
61                        bail!(
62                            "query request returned non-success status code: {}",
63                            res.status()
64                        )
65                    }
66                }
67            };
68            let query = format!(
69                "{:#}",
70                rsexpr::from_slice_multi(&query).map_err(|errs| anyhow!(errs
71                    .into_iter()
72                    .map(|err| err.to_string())
73                    .collect::<Vec<_>>()
74                    .join(", ")))?
75            );
76            fs::write(file.path().with_file_name(format!("new.{filename}")), query)?;
77        }
78    }
79
80    // look for missing queries
81    let langs_toml_path = crate::WORKSPACE_DIR.join("syntastica-macros/languages.toml");
82    let mut langs_toml = fs::read_to_string(&langs_toml_path)?;
83    for lang in &crate::LANGUAGE_CONFIG.languages {
84        let kinds = match (lang.queries.injections, lang.queries.locals) {
85            (false, false) => &["injections", "locals"][..],
86            (false, true) => &["injections"],
87            (true, false) => &["locals"],
88            (true, true) => &[],
89        };
90
91        for &kind in kinds {
92            let queries = fetch_query(&lang.name, kind)?;
93            if let Some(text) = queries {
94                fs::write(
95                    crate::WORKSPACE_DIR.join(format!("queries/{}/{kind}.scm", lang.name)),
96                    text,
97                )?;
98                println!("found new {kind} queries for {}", lang.name);
99
100                let (before, rest) = langs_toml
101                    .split_once(&format!("\nname = \"{}\"", lang.name))
102                    .expect("language should be lang config");
103                if kind == "injections" {
104                    langs_toml = format!(
105                        "{before}\nname = \"{}\"{}",
106                        lang.name,
107                        rest.replacen("\ninjections = false", "\ninjections = true", 1),
108                    );
109                } else if kind == "locals" {
110                    langs_toml = format!(
111                        "{before}\nname = \"{}\"{}",
112                        lang.name,
113                        rest.replacen("\nlocals = false", "\nlocals = true", 1),
114                    );
115                }
116            }
117        }
118    }
119    fs::write(&langs_toml_path, langs_toml)?;
120
121    Ok(())
122}

134pub fn fetch_query(name: &str, kind: &str) -> Result<Option<String>> {
135    if kind == "highlights" {
136        if let Some(query) = read_mapped_highlights(name)? {
137            return Ok(Some(query));
138        }
139    } else if kind == "injections" || kind == "locals" {
140        if let Some(query) = read_mapped_sibling_local(name, kind)? {
141            return Ok(Some(query));
142        }
143    }
144
145    if let Some(query) = read_local_query(name, kind)? {
146        return Ok(Some(forked_from(
147            name,
148            kind,
149            &format_query(&query).context("failed to parse downloaded queries")?,
150        )));
151    }
152
153    // Try both old and new URL patterns
154    let urls = [
155        format!("https://raw.githubusercontent.com/nvim-treesitter/nvim-treesitter/main/runtime/queries/{name}/{kind}.scm"),
156        format!("https://raw.githubusercontent.com/nvim-treesitter/nvim-treesitter/master/queries/{name}/{kind}.scm"),
157    ];
158
159    for url in urls {
160        if let Ok(res) = reqwest::blocking::get(&url) {
161            if res.status().is_success() {
162                if let Ok(query) = res.text() {
163                    return Ok(Some(forked_from(
164                        name,
165                        kind,
166                        &format_query(&query).context("failed to parse downloaded queries")?,
167                    )));
168                }
169            }
170        }
171    }
172
173    Ok(None)
174}

176fn format_query(query: &str) -> Result<String> {
177    Ok(format!(
178        "{:#}",
179        rsexpr::from_slice_multi(query).map_err(|errs| anyhow!(errs
180            .into_iter()
181            .map(|err| err.to_string())
182            .collect::<Vec<_>>()
183            .join(", ")))?
184    ))
185}

187fn read_local_query(name: &str, kind: &str) -> Result<Option<String>> {
188    let Some(root) = local_nvim_treesitter_root() else {
189        return Ok(None);
190    };
191
192    let runtime_path = root
193        .join("runtime/queries")
194        .join(name)
195        .join(format!("{kind}.scm"));
196    if runtime_path.exists() {
197        return Ok(Some(fs::read_to_string(runtime_path)?));
198    }
199
200    let legacy_path = root.join("queries").join(name).join(format!("{kind}.scm"));
201    if legacy_path.exists() {
202        return Ok(Some(fs::read_to_string(legacy_path)?));
203    }
204
205    Ok(None)
206}

208fn local_nvim_treesitter_root() -> Option<PathBuf> {
209    if let Ok(path) = env::var("NVIM_TREESITTER_PATH").or_else(|_| env::var("NVIM_TREESITTER_REPO"))
210    {
211        let root = PathBuf::from(path);
212        if root.exists() {
213            return Some(root);
214        }
215    }
216    let home = env::var("HOME").ok()?;
217    let root = PathBuf::from(home).join("github/nvim-treesitter/nvim-treesitter");
218    if root.exists() {
219        Some(root)
220    } else {
221        None
222    }
223}

243fn read_mapped_highlights(name: &str) -> Result<Option<String>> {
244    let Some(mapping) = find_grammar_mapping(name)? else {
245        return Ok(None);
246    };
247    let Some(path) = mapping
248        .highlights_scm_path
249        .as_deref()
250        .map(str::trim)
251        .filter(|p| !p.is_empty())
252    else {
253        return Ok(None);
254    };
255
256    if let Some(text) = read_mapped_file_local(&mapping, path)? {
257        return Ok(Some(text));
258    }
259
260    if let Some(text) = read_mapped_highlights_remote(&mapping, path)? {
261        return Ok(Some(text));
262    }
263
264    Ok(None)
265}

267fn read_mapped_sibling_local(name: &str, kind: &str) -> Result<Option<String>> {
268    let Some(mapping) = find_grammar_mapping(name)? else {

290fn read_mapped_file_local(mapping: &GrammarMapping, path: &str) -> Result<Option<String>> {
291    let Some(root) = local_grammar_root() else {
292        return Ok(None);
293    };
294
295    let rel_path = path.trim_start_matches('/');
296    let repo_root = root.join(&mapping.grammar);
297    let mut candidates = Vec::new();
298    candidates.push(repo_root.join(rel_path));
299    if !mapping.grammar_rev.trim().is_empty() {
300        candidates.push(repo_root.join(&mapping.grammar_rev).join(rel_path));
301    }
302
303    for candidate in candidates {
304        if candidate.is_file() {
305            let content = fs::read_to_string(&candidate)
306                .with_context(|| format!("failed reading {}", candidate.display()))?;
307            let formatted = format_query(&content).context("failed to parse downloaded queries")?;
308            let url = mapped_highlights_url(mapping, rel_path)
309                .unwrap_or_else(|| format!("file://{}", candidate.to_string_lossy().into_owned()));
310            return Ok(Some(forked_from_url(&url, &formatted)));
311        }
312    }
313
314    Ok(None)
315}

317fn read_mapped_highlights_remote(mapping: &GrammarMapping, path: &str) -> Result<Option<String>> {
318    let Some(url) = mapped_highlights_url(mapping, path.trim_start_matches('/')) else {
319        return Ok(None);
320    };
321
322    let res = match reqwest::blocking::get(&url) {
323        Ok(res) => res,
324        Err(_) => return Ok(None),
325    };
326    if !res.status().is_success() {
327        return Ok(None);
328    }
329    let query = res.text()?;
330    let formatted = format_query(&query).context("failed to parse downloaded queries")?;
331    Ok(Some(forked_from_url(&url, &formatted)))
332}

334fn mapped_highlights_url(mapping: &GrammarMapping, rel_path: &str) -> Option<String> {
335    let (repo, rev) = match mapping.highlights_scm_repo.as_ref() {
336        Some(repo) => {
337            let Some(rev) = mapping
338                .highlights_scm_ref
339                .as_deref()
340                .map(str::trim)
341                .filter(|r| !r.is_empty())
342            else {
343                return None;
344            };
345            (repo, rev)
346        }
347        None => (mapping.grammar_repo.as_ref()?, mapping.grammar_rev.as_str()),
348    };
349    let content_url = crate::add_lang::url_to_content_url(repo, rev)?;
350    Some(format!(
351        "{}/{}",
352        content_url.trim_end_matches('/'),
353        rel_path.trim_start_matches('/')
354    ))
355}

357fn find_grammar_mapping(name: &str) -> Result<Option<&'static GrammarMapping>> {
358    let mappings = grammar_mappings()?;
359    let name = name.trim();
360    if name.is_empty() {
361        return Ok(None);
362    }
363
364    let mut candidates = vec![name.to_string()];
365    if let Some(stripped) = name.strip_suffix("_only") {
366        candidates.push(stripped.to_string());
367    }
368
369    for candidate in candidates {
370        let candidate_sanitized = sanitize_lang_name(&candidate);
371        if let Some(mapping) = mappings.iter().find(|m| {
372            m.grammar == candidate
373                || sanitize_lang_name(&m.grammar) == candidate_sanitized
374                || m.effective_filetype.as_deref() == Some(candidate.as_str())
375                || m.nvim_filetype.as_deref() == Some(candidate.as_str())
376        }) {
377            return Ok(Some(mapping));
378        }
379    }
380
381    Ok(None)
382}

388fn grammar_mappings() -> Result<&'static [GrammarMapping]> {
389    static MAPPINGS: Lazy<Result<Vec<GrammarMapping>>> = Lazy::new(|| {
390        let path = grammar_mappings_path();
391        let content = fs::read_to_string(&path)
392            .with_context(|| format!("failed to read grammar mappings at {}", path.display()))?;
393        let mappings: Vec<GrammarMapping> =
394            serde_json::from_str(&content).context("invalid grammar mappings json")?;
395        Ok(mappings)
396    });
397    match MAPPINGS.as_ref() {
398        Ok(mappings) => Ok(mappings.as_slice()),
399        Err(err) => Err(anyhow!("unable to load grammar mappings: {err:?}")),
400    }
401}

413fn sanitize_lang_name(name: &str) -> String {
414    let mut out = String::with_capacity(name.len());
415    for ch in name.chars() {
416        if ch.is_ascii_alphanumeric() || ch == '_' {
417            out.push(ch.to_ascii_lowercase());
418        } else {
419            out.push('_');
420        }
421    }
422    if out.chars().next().is_some_and(|c| c.is_ascii_digit()) {
423        out.insert(0, '_');
424    }
425    out
