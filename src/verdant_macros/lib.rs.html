16#[derive(Debug)]
17struct FfiGroup<'a> {
18    representative: &'a Language,
19    aliases: Vec<&'a Language>,
20}
21
22fn ffi_groups(languages: &[Language]) -> Vec<FfiGroup<'_>> {
23    let mut group_indices: std::collections::HashMap<&str, usize> =
24        std::collections::HashMap::new();
25    let mut groups: Vec<FfiGroup<'_>> = Vec::new();
26
27    for lang in languages {
28        if let Some(&idx) = group_indices.get(lang.parser.ffi_func.as_str()) {
29            groups[idx].aliases.push(lang);
30        } else {
31            group_indices.insert(lang.parser.ffi_func.as_str(), groups.len());
32            groups.push(FfiGroup {
33                representative: lang,
34                aliases: vec![lang],
35            });
36        }
37    }
38
39    groups
40}

42fn not_wasm_cfg(lang: &Language) -> proc_macro2::TokenStream {
43    let raw_wasm_cfg = quote! { target_family = "wasm" };
44    let raw_wasm_unknown_cfg = quote! { all(target_arch = "wasm32", target_vendor = "unknown", target_os = "unknown", target_env = "") };
45    match (
46        lang.wasm,
47        lang.parser.external_scanner.cpp || !lang.wasm_unknown,
48    ) {
49        (false, _) => quote! { , not(#raw_wasm_cfg) },
50        (_, true) => quote! { , not(#raw_wasm_unknown_cfg) },
51        _ => quote! {},
52    }
53}

55fn lang_feature_target_cfg(lang: &Language) -> proc_macro2::TokenStream {
56    let name_str = &lang.name;
57    let not_wasm_cfg = not_wasm_cfg(lang);
58    quote! { all(feature = #name_str #not_wasm_cfg) }
59}

61fn ffi_group_cfg(group: &FfiGroup<'_>) -> proc_macro2::TokenStream {
62    let alias_cfgs = group
63        .aliases
64        .iter()
65        .map(|lang| lang_feature_target_cfg(lang))
66        .collect::<Vec<_>>();
67
68    match alias_cfgs.as_slice() {
69        [cfg] => cfg.clone(),
70        _ => quote! { any(#(#alias_cfgs),*) },
71    }
72}

74#[proc_macro]
75pub fn parsers_git(_: TokenStream) -> TokenStream {
76    ffi_groups(&LANGUAGE_CONFIG.languages)
77        .iter()
78        .map(|group| {
79            // The first language keeps the canonical build metadata for the shared FFI symbol;
80            // additional aliases only extend the feature gate.
81            let lang = group.representative;
82            let cfg = ffi_group_cfg(group);
83            let name = &lang.name;
84            let url = &lang.parser.git.url;
85            let rev = &lang.parser.git.rev;
86            let external_c = lang.parser.external_scanner.c;
87            let external_cpp = lang.parser.external_scanner.cpp;
88            let path = match &lang.parser.git.path {
89                Some(path) => quote_use! { Some(#path) },
90                None => quote_use! { None },
91            };
92            let wasm = lang.wasm;
93            let wasm_unknown = lang.wasm_unknown;
94            let generate = lang.parser.generate;
95            quote! {
96                #[cfg(#cfg)]
97                compile_parser(#name, #url, #rev, #external_c, #external_cpp, #path, #wasm, #wasm_unknown, #generate)?;
98            }
99        })
100        .collect::<proc_macro2::TokenStream>()
101        .into()
102}

104#[proc_macro]
105pub fn parsers_ffi(_: TokenStream) -> TokenStream {
106    let ffi_groups = ffi_groups(&LANGUAGE_CONFIG.languages);
107    let extern_c = ffi_groups.iter().map(|group| {
108        let ffi_func = format_ident!("{}", group.representative.parser.ffi_func);
109        let cfg = ffi_group_cfg(group);
110        quote! {
111            #[cfg(#cfg)]
112            fn #ffi_func() -> ::verdant_core::language_set::Language;
113        }
114    });
115    let functions = LANGUAGE_CONFIG.languages.iter().map(|lang| {
116        let feat = lang.group.to_string();
117        let name = format_ident!("{}", lang.name);
118        let name_str = &lang.name;
119        let ffi_func = format_ident!("{}", lang.parser.ffi_func);
120        let doc = format!(
121            "Get the parser for [{}]({}/tree/{}). {}",
122            lang.name,
123            lang.parser.git.url,
124            lang.parser.git.rev,
125            match (
126                lang.wasm,
127                lang.parser.external_scanner.cpp || !lang.wasm_unknown
128            ) {
129                (false, _) => "(not supported on WebAssembly targets)",
130                (_, true) => "(not supported on the `wasm32-unknown-unknown` target)",
131                _ => "",
132            },
133        );
134        // disable unsupported parsers on wasm
135        let not_wasm_cfg = not_wasm_cfg(lang);
136        quote! {
137            #[cfg(all(any(feature = #feat, feature = #name_str) #not_wasm_cfg))]
138            #[doc = #doc]
139            pub fn #name() -> ::verdant_core::language_set::Language {
140                #[cfg(not(all(feature = "docs", doc)))]
141                unsafe { #ffi_func() }
142                #[cfg(all(feature = "docs", doc))]
143                ::std::unimplemented!()
144            }
145        }
146    });
147    parsers(
148        "verdant_parsers_git",
149        functions,
150        |_| true,
151        Some(quote! {
152            #[cfg(not(all(feature = "docs", doc)))]
153            extern "C" {
154                #(#extern_c)*
155            }
156        }),
157        "",
158    )
159}

171fn parsers_rust(crate_name: &str, crates_io: bool, query_suffix: &str) -> TokenStream {
172    let functions = LANGUAGE_CONFIG.languages.iter().map(|lang| {
173        let feat = lang.group.to_string();
174        let name = format_ident!("{}", lang.name);
175        let name_str = &lang.name;
176        let (doc, body) = match (&lang.parser.rust_const, &lang.parser.rust_func) {
177            (Some(ident), _) if lang.parser.supports(!crates_io) => {
178                let ident = format_ident!("{ident}");
179                let package = format_ident!("{}", lang.parser.package.replace('-', "_"));
180                (
181                    format!(
182                        "Get the parser for [{}]({}/tree/{}).",
183                        lang.name, lang.parser.git.url, lang.parser.git.rev,
184                    ),
185                    quote! { ::verdant_core::language_set::Language::new(#package::#ident) }
186                )
187            },
188            (_, Some(func)) if lang.parser.supports(!crates_io) => {
189                let func = format_ident!("{func}");
190                let package = format_ident!("{}", lang.parser.package.replace('-', "_"));
191                (
192                    format!(
193                        "Get the parser for [{}]({}/tree/{}).",
194                        lang.name, lang.parser.git.url, lang.parser.git.rev,
195                    ),
196                    quote! { #package::#func() }
197                )
198            },
199            _ => (
200                "**This parser is not supported by this parser collection and thus this function will panic!**"
201                    .to_owned(),
202                quote! { ::std::unimplemented!() }
203            ),
204        };
205        quote! {
206            #[cfg(any(feature = #feat, feature = #name_str))]
207            #[doc = #doc]
208            pub fn #name() -> ::verdant_core::language_set::Language {
209                #body
210            }
211        }
212    });
213    parsers(
214        crate_name,
215        functions,
216        |lang| lang.parser.supports(!crates_io),
217        None,
218        query_suffix,
219    )
220}

222fn parsers(
223    crate_name: &str,
224    functions: impl Iterator<Item = proc_macro2::TokenStream>,
225    filter: impl Fn(&&Language) -> bool,
226    extra: Option<proc_macro2::TokenStream>,
227    query_suffix: &str,
228) -> TokenStream {
229    let list = LANGUAGE_CONFIG
230        .languages
231        .iter()
232        .filter(&filter)
233        .map(|lang| {
234            let name_str = &lang.name;
235            let variant = format_ident!("{}", lang.name.to_pascal_case());
236            let not_wasm_cfg = not_wasm_cfg(lang);
237            quote! { #[cfg(all(feature = #name_str #not_wasm_cfg))] Lang::#variant }
238        });
239    let names_list = LANGUAGE_CONFIG
240        .languages
241        .iter()
242        .filter(&filter)
243        .map(|lang| {
244            let name_str = &lang.name;
245            let not_wasm_cfg = not_wasm_cfg(lang);
246            quote! { #[cfg(all(feature = #name_str #not_wasm_cfg))] #name_str }
247        });
248    let file_types = LANGUAGE_CONFIG
249        .languages
250        .iter()
251        .filter(&filter)
252        .flat_map(|lang| {
253            let name_str = &lang.name;
254            let variant = format_ident!("{}", lang.name.to_pascal_case());
255            let not_wasm_cfg = not_wasm_cfg(lang);
256            lang.file_types.iter().map(move |ft| {
257                let ft = format_ident!("{ft:?}");
258                quote! {
259                    #[cfg(all(feature = #name_str #not_wasm_cfg))]
260                    (::verdant_core::language_set::FileType::#ft, Lang::#variant)
261                }
262            })
263        });
264    let mut langs_sorted_by_group = LANGUAGE_CONFIG.languages.clone();
265    langs_sorted_by_group.sort_by_key(|lang| lang.group);
266    let func_map = langs_sorted_by_group.iter().filter(&filter).map(|lang| {
267        let name_str = &lang.name;
268        let variant = format_ident!("{}", lang.name.to_pascal_case());
269        let not_wasm_cfg = not_wasm_cfg(lang);
270        quote! {
271            #[cfg(all(feature = #name_str #not_wasm_cfg))]
272            {
273                _map.insert(Lang::#variant, _idx);
274                _idx += 1;
275            }
276        }
277    });
278    let funcs = langs_sorted_by_group.iter().filter(&filter).map(|lang| {
279        let name = format_ident!("{}", lang.name);
280        let name_str = &lang.name;
281        let not_wasm_cfg = not_wasm_cfg(lang);
282        quote! { #[cfg(all(feature = #name_str #not_wasm_cfg))] #name }
283    });
284    let queries = langs_sorted_by_group.iter().filter(&filter).map(|lang| {
285        let name_str = &lang.name;
286        let highlights = format_ident!("{}_HIGHLIGHTS{query_suffix}", lang.name.to_uppercase());
287        let injections = format_ident!("{}_INJECTIONS{query_suffix}", lang.name.to_uppercase());
288        let locals = format_ident!("{}_LOCALS{query_suffix}", lang.name.to_uppercase());
289        let not_wasm_cfg = not_wasm_cfg(lang);
290
291        quote! { #[cfg(all(feature = #name_str #not_wasm_cfg))] [
292            ::verdant_queries::#highlights,
293            ::verdant_queries::#injections,
294            ::verdant_queries::#locals,
295        ] }
296    });
297    let lang_enum_example_use = format!("use {crate_name}::{{Lang, LANGUAGES, LANGUAGE_NAMES}};");
298    let lang_enum = LANGUAGE_CONFIG
299        .languages
300        .iter()
301        .filter(&filter)
302        .map(|lang| {
303            let feat = lang.group.to_string();
304            let name_str = &lang.name;
305            let ft_support = if lang.file_types.is_empty() {
306                Cow::Borrowed("supports no file types")
307            } else {
308                format!(
309                    "supports these file types: {}",
310                    lang.file_types
311                        .iter()
312                        .map(|ft| format!(
313                            "[`{ft}`](::verdant_core::language_set::FileType::{ft:?})"
314                        ))
315                        .collect::<Vec<_>>()
316                        .join(", ")
317                )
318                .into()
319            };
320            let doc = format!("Provides the [`{name_str}`] language, {ft_support}.");
321            let variant = format_ident!("{}", lang.name.to_pascal_case());
322            let not_wasm_cfg = not_wasm_cfg(lang);
323            quote! {
324                #[doc = #doc]
325                #[cfg(all(any(feature = #feat, feature = #name_str) #not_wasm_cfg))]
326                #variant
327            }
328        });
329    let lang_get_match = LANGUAGE_CONFIG
330        .languages
331        .iter()
332        .filter(&filter)
333        .map(|lang| {
334            let name = format_ident!("{}", lang.name);
335            let name_str = &lang.name;
336            let variant = format_ident!("{}", lang.name.to_pascal_case());
337            let not_wasm_cfg = not_wasm_cfg(lang);
338            quote! {
339                #[cfg(all(feature = #name_str #not_wasm_cfg))]
340                Self::#variant => #name(),
341            }
342        });
343    let lang_set_type = quote! { type Language = Lang; };
344    let cfg_test = quote! { #[cfg(test)] };
345    let lang_tests = LANGUAGE_CONFIG
346        .languages
347        .iter()
348        .filter(&filter)
349        .map(|lang| {
350            let name = format_ident!("{}", lang.name);
351            let name_str = &lang.name;
352            let variant = format_ident!("{}", lang.name.to_pascal_case());
353            let not_wasm_cfg = not_wasm_cfg(lang);
354            quote! {
355                #[test]
356                #[cfg(all(feature = #name_str #not_wasm_cfg))]
357                fn #name() {
358                    assert_eq!(crate::#name(), crate::Lang::#variant.get());
359                    let set = crate::LanguageSetImpl::new();
360                    let result = ::verdant_core::language_set::LanguageSet::get_language(
361                        &set,
362                        crate::Lang::#variant,
363                    );
364                    assert!(
365                        result.is_ok(),
366                        "get_language failed for {}: {:?}",
367                        #name_str,
368                        result.as_ref().err()
369                    );
370                }
371            }
372        });
373
374    quote_use! {
375        # use std::{borrow::Cow, collections::HashMap};
376
377        # use verdant_core::{
378            language_set::{HighlightConfiguration, LanguageSet, Language, FileType, SupportedLanguage},
379            Error, Result,
380            theme::THEME_KEYS,
381        };
382        # use once_cell::sync::{Lazy, OnceCell};
383
384        #extra
385
386        /// A list of all languages supported by the current feature set.
387        pub const LANGUAGES: &[Lang] = &[#(#list),*];
388        const LANG_COUNT: usize = LANGUAGES.len();
389
390        /// A list of all language names supported by the current feature set.
391        pub const LANGUAGE_NAMES: &[&str] = &[#(#names_list),*];
392
393        #(#functions)*
394
395        // TODO: use "perfect" hashmap with compile-time known keys
396        static FILE_TYPE_MAP: Lazy<HashMap<FileType, Lang>>
397            = Lazy::new(|| HashMap::from([#(#file_types),*]));
398
399        // TODO: use "perfect" hashmap with compile-time known keys
400        static IDX_MAP: Lazy<HashMap<Lang, usize>> = Lazy::new(|| {
401            let mut _map = HashMap::new();
402            let mut _idx = 0;
403            #(#func_map)*
404            _map
405        });
406
407        const QUERIES: &[[&str; 3]] = &[#(#queries),*];
408        const FUNCS: &[fn() -> Language] = &[#(#funcs),*];
409
410        /// An enum of every supported language in the current feature set.
411        ///
412        /// An instance of the respective tree-stter
413        /// [`Language`](::verdant_core::language_set::Language) can be obtained with the
414        /// [`get`](Lang::get) method.
415        ///
416        /// You can also get a [`Lang`] from its name using
417        /// [`for_name`](::verdant_core::language_set::SupportedLanguage::for_name), or for a
418        /// [`FileType`](::verdant_core::language_set::FileType) using
419        /// [`for_file_type`](::verdant_core::language_set::SupportedLanguage::for_file_type).
420        /// See the docs for each variant to see its "name" and the supported file types.
421        /// Both of these require the
422        /// [`SupportedLanguage`](::verdant_core::language_set::SupportedLanguage) trait to be
423        /// in scope.
424        ///
425        /// See [`LANGUAGES`] for a list containing all variants and [`LANGUAGE_NAMES`] for a list
426        /// of all valid names.
427        ///
428        /// The enum is marked as non-exhaustive for two reasons:
429        ///
430        /// 1. New languages may be added in the future
431        /// 2. The variants are enabled/disabled by features
432        ///
433        /// # Example
434        ///
435        /// ```
436        #[doc = #lang_enum_example_use]
437        /// use verdant_core::language_set::{SupportedLanguage, FileType};
438        ///
439        /// // you can get a `Lang` from its name
440        /// assert_eq!(Lang::Rust, Lang::for_name("rust", &()).unwrap());
441        /// // and for a file type
442        /// assert_eq!(Some(Lang::Rust), Lang::for_file_type(FileType::Rust, &()));
443        ///
444        /// // `LANGUAGES` is a list of all variants,
445        /// // `LANGUAGE_NAMES` is a list of all variant names
446        /// for (&lang, &name) in LANGUAGES.iter().zip(LANGUAGE_NAMES) {
447        ///     assert_eq!(lang, Lang::for_name(name, &()).unwrap());
448        ///
449        ///     // `Lang` instances can be turned into strings
450        ///     assert_eq!(lang, Lang::for_name(<Lang as SupportedLanguage<'_, ()>>::name(&lang), &()).unwrap());
451        ///     assert_eq!(lang, Lang::for_name(lang.to_string(), &()).unwrap());
452        ///     assert_eq!(lang, Lang::for_name(lang.as_ref(), &()).unwrap());
453        ///     let lang_name: &'static str = lang.into();
454        ///     assert_eq!(lang, Lang::for_name(lang_name, &()).unwrap());
455        /// }
456        /// ```
457        #[non_exhaustive]
458        #[derive(
459            Debug,
460            Clone,
461            Copy,
462            PartialEq,
463            Eq,
464            Hash,
465            ::strum::Display,
466            ::strum::AsRefStr,
467            ::strum::IntoStaticStr,
468            ::strum::EnumString,
469        )]
470        #[strum(serialize_all = "snake_case", use_phf)]
471        pub enum Lang {
472            #(#lang_enum),*
473        }
474
475        impl Lang {
476            /// Get an instance of the respective
477            /// [`Language`](::verdant_core::language_set::Language).
478            pub fn get(&self) -> Language {
479                match self {
480                    #(#lang_get_match)*
481                    _ => unreachable!("all variants are matched")
482                }
483            }
484
485            /// Create an instance of the corresponding
486            /// [`HighlightConfiguration`](::verdant_core::language_set::HighlightConfiguration).
487            pub fn get_config(&self) -> Result<HighlightConfiguration> {
488                let idx = IDX_MAP[self];
489                let lang = FUNCS[idx]();
490                let mut conf = HighlightConfiguration::new(
491                    lang,
492                    LANGUAGE_NAMES[idx],
493                    QUERIES[idx][0],
494                    QUERIES[idx][1],
495                    QUERIES[idx][2],
496                )?;
497                conf.configure(THEME_KEYS);
498                Ok(conf)
499            }
500
501            /// Get the highlights query for this language.
502            pub fn highlights_query(&self) -> &'static str {
503                let idx = IDX_MAP[self];
504                QUERIES[idx][0]
505            }
506
507            /// Get the injections query for this language.
508            pub fn injections_query(&self) -> &'static str {
509                let idx = IDX_MAP[self];
510                QUERIES[idx][1]
511            }
512
513            /// Get the locals query for this language.
514            pub fn locals_query(&self) -> &'static str {
515                let idx = IDX_MAP[self];
516                QUERIES[idx][2]
517            }
518        }
519
520        impl<S> SupportedLanguage<'_, S> for Lang {
521            fn name(&self) -> Cow<'_, str> {
522                Cow::Borrowed(self.into())
523            }
524
525            fn for_name(name: impl AsRef<str>, _set: &S) -> Result<Self> {
526                <Self as ::std::str::FromStr>::from_str(name.as_ref())
527                    .map_err(|_| Error::UnsupportedLanguage(name.as_ref().to_owned()))
528            }
529
530            fn for_file_type(file_type: FileType, _set: &S) -> Option<Self> {
531                FILE_TYPE_MAP
532                    .get(&file_type)
533                    .map(|name| (*name).into())
534            }
535        }
536
537        /// An implementation of [`LanguageSet`](::verdant_core::language_set::LanguageSet)
538        /// including all languages in the enabled feature set.
539        ///
540        /// Languages are loaded the first time they are requested and will then be reused for
541        /// later accesses. To pre-load a list of languages, use
542        /// [`preload`](LanguageSetImpl::preload) or [`preload_all`](LanguageSetImpl::preload_all).
543        pub struct LanguageSetImpl([OnceCell<HighlightConfiguration>; LANG_COUNT]);
544
545        impl LanguageSet<'_> for LanguageSetImpl {
546            #lang_set_type
547
548            fn get_language(&self, language: Self::Language) -> Result<&HighlightConfiguration> {
549                let idx = IDX_MAP[&language];
550                self.0[idx].get_or_try_init(|| language.get_config())
551            }
552        }
553
554        impl LanguageSetImpl {
555            /// Create a new [`LanguageSetImpl`] with no pre-loaded languages.
556            pub fn new() -> Self {
557                #[allow(clippy::declare_interior_mutable_const)]
558                const INIT: OnceCell<HighlightConfiguration> = OnceCell::new();
559                Self([INIT; LANG_COUNT])
560            }
561
562            /// Pre-load the given list of languages.
563            ///
564            /// To pre-load all supported languages, use [`preload_all`](LanguageSetImpl::preload_all).
565            pub fn preload(&self, languages: &[Lang]) -> Result<()> {
566                for lang in languages {
567                    let idx = IDX_MAP[lang];
568                    let entry = &self.0[idx];
569                    if entry.get().is_none() {
570                        drop(entry.set(lang.get_config()?));
571                    }
572                }
573                Ok(())
574            }
575
576            /// Pre-load all languages in this set.
577            ///
578            /// To pre-load a specific set of languages, use [`preload`](LanguageSetImpl::preload).
579            pub fn preload_all(&self) -> Result<()> {
580                self.preload(LANGUAGES)
581            }
582        }
583
584        impl Default for LanguageSetImpl {
585            fn default() -> Self {
586                Self::new()
587            }
588        }

591        mod tests {
592            #(#lang_tests)*
593        }
594    }
595    .into()
596}

769#[cfg(feature = "js")]
770#[proc_macro]
771pub fn js_lang_lib(input: TokenStream) -> TokenStream {
772    let lang_name = syn::parse_macro_input!(input as syn::LitStr).value();
773    let lang = LANGUAGE_CONFIG
774        .languages
775        .iter()
776        .find(|lang| lang.name == lang_name)
777        .unwrap_or_else(|| panic!("language '{lang_name}' is not defined"));
778
779    let func = format_ident!("verdant_lang_{lang_name}");
780    let ffi_func = format_ident!("{}", &lang.parser.ffi_func);
781    let name = std::ffi::CString::new(lang_name.as_str()).unwrap();
782    let file_types = lang
783        .file_types
784        .iter()
785        .map(|ft| std::ffi::CString::new(ft.as_ref()).unwrap());
786    let highlights = format_ident!("{}_HIGHLIGHTS", lang_name.to_uppercase());
787    let injections = format_ident!("{}_INJECTIONS", lang_name.to_uppercase());
788    let locals = format_ident!("{}_LOCALS", lang_name.to_uppercase());
789
790    quote_use! {
791        #use core::ffi::{c_char, c_void};
792
793        extern "C" {
794            fn malloc(size: usize) -> *mut c_void;
795            fn memcpy(dest: *mut c_void, src: *const c_void, count: usize) -> *mut c_void;
796            fn #ffi_func() -> Language;
797        }
798
799        fn str_to_cstr(str: &'static str) -> *const c_char {
800            let ptr = unsafe { malloc(str.len() + 1) };
801            unsafe { memcpy(ptr, str.as_ptr() as *const _, str.len()) };
802            unsafe { (ptr as *mut c_char).add(str.len()).write(0) }
803            ptr as *const _
804        }
805
806        #[no_mangle]
807        pub fn #func() -> *mut LangInfo {
808            let ptr = unsafe { malloc(::core::mem::size_of::<LangInfo>()) } as *mut LangInfo;
809            const NAME: *const c_char = #name.as_ptr();
810            const FILE_TYPES: &[*const c_char] = &[#(#file_types.as_ptr()),*];
811            let info = LangInfo {
812                name: NAME,
813                file_types: FILE_TYPES.as_ptr(),
814                file_types_len: FILE_TYPES.len(),
815                language: unsafe { #ffi_func() },
816                highlights_query: str_to_cstr(::verdant_queries::#highlights),
817                injections_query: str_to_cstr(::verdant_queries::#injections),
818                locals_query: str_to_cstr(::verdant_queries::#locals),
819            };
820            unsafe { ptr.write(info) };
821            ptr
822        }
823    }
824    .into()
825}

830    let lang_name = syn::parse_macro_input!(input as syn::LitStr).value();

846    let wasm_unknown = lang.wasm_unknown;
847    let generate = lang.parser.generate;
848    quote! {
849        compile_parser(#lang_name, #url, #rev, #external_c, #external_cpp, #path, #wasm, #wasm_unknown, #generate)?;
850    }
